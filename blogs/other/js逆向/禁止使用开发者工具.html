禁止使用开发者工具

绝对零度 2024/4/8 0 js逆向反调试

我们在进行js逆向的时候需要打开浏览器开发者工具进行网页的调试，但是在有时候网站会阻止我们去调试它的代码，而其阻止我们的手段也是多种多样的。

这里的一段代码禁止了F12、鼠标右键打开开发者工具等功能

var debugflag = false;
document.onkeydown = function() {
  if ((e.ctrlKey) && (e.keyCode == 83)) {
    alert("检测到非法调试，CTRL + S被管理员禁用");
    return false;
  }
}
document.onkeydown = function() {
  var e = window.event || arguments[0];
  if (e.keyCode == 123) {
    alert("检测到非法调试，F12被管理员禁用");
    return false;
  }
}
document.oncontextmenu = function() {
  alert('检测到非法调试，右键被管理员禁用');
  return false;
}
